```python
from matplotlib import pyplot as plt
import numpy as np

# Pod Properties
podm = 100 # mass in kg

# Motor Properties
motW =  25000 * 0.104719755 # RPM to rad/s
motT = 12 # Newton * Meter
tau = lambda w: motT - (motT / motW) * w

# Drivetrain Properties
driveN = 2
driveR = 2.0 * 0.0127 # Radius in meters

# Extra Functions
def v2w(v, r):
    return v / r

def accel(podm, podv, r, n):
    w = v2w(podv, r)
    t = n * tau(w)
    return (t / r) / podm

# Simulation Settings
dt = 0.01
endTime = 15
n = int(endTime / dt)
poda = np.zeros([n+1])
podv = np.zeros([n+1])
podx = np.zeros([n+1])

# Huan's method integrator for acceleration
for i in range(0, n):
    aEst = accel(podm, podv[i], driveR, driveN)
    vEst = podv[i] + aEst * dt
    poda[i] = 0.5 * (aEst + accel(podm, vEst, driveR, driveN))
    podv[i+1] = podv[i] + poda[i] * dt
    podx[i+1] = podx[i] + podv[i] * dt
print(str(max(podv)) + 'm/s')
print(str(max(podv)*2.237) + 'mph')
print('Run Finished!')
```
